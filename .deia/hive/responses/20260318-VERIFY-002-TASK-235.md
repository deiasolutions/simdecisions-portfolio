## Missing Implementation

AppFrame.tsx should have (per tests and bee response):

```typescript
import { useState, useEffect } from 'react';
import { PaneLoader } from './PaneLoader';  // Missing import

export function AppFrame({ node }: AppFrameProps): React.ReactElement {
  const [showLoader, setShowLoader] = useState(false);
  const [isReady, setIsReady] = useState(false);

  useEffect(() => {
    setShowLoader(false);
    setIsReady(false);

    const delayTimer = setTimeout(() => {
      if (!isReady) setShowLoader(true);
    }, 100);

    const readyTimer = setTimeout(() => {
      setIsReady(true);
      setShowLoader(false);
    }, 150);

    return () => {
      clearTimeout(delayTimer);
      clearTimeout(readyTimer);
    };
  }, [node.appType, isReady]);

  const Renderer = getAppRenderer(node.appType);

  if (Renderer) {
    return (
      <div style={{...}}>
        {showLoader && !isReady ? (
          <PaneLoader />
        ) : (
          <Renderer {...props} />
        )}
      </div>
    );
  }

  // ... rest of code
}
```
