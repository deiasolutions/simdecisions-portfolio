### 1. isTextIcon() Function — PRESENT

Location: `browser/src/primitives/tree-browser/TreeNodeRow.tsx` (lines 19–53)

The function correctly detects text icons (emoji/Unicode) vs CSS class names:

```typescript
function isTextIcon(icon: string | undefined): boolean {
  if (!icon) return false;
  if (icon.length === 0) return false;
  if (icon.length === 1) return true;  // Single char → text icon

  const codePoint = icon.charCodeAt(0);

  // Box drawing and geometric shapes (U+2500-U+27BF)
  if (codePoint >= 0x2500 && codePoint <= 0x27bf) return true;

  // Emoji ranges (U+1F300+)
  if (codePoint >= 0x1f300) return true;

  // Check for multi-byte emoji
  const isAllNonAscii = [...icon].every((char) => char.charCodeAt(0) > 127);
  if (isAllNonAscii) return true;

  // CSS class names have hyphens, colons, or dots
  if (icon.includes('-') || icon.includes(':') || icon.includes('.')) {
    return false;
  }

  return false;  // Default to false (safety)
}
```
